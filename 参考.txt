    // æµå¼å“åº”å¤„ç†å‡½æ•°
    async function fetchData() {
        button.disabled = true;
        let aiMessageElement = null;
        let accumulatedResponse = '';

        try {
            const requestData = {
                "last_prompt": content[content.length - 1].message,
                "conversation_history": content.map(item => ({
                    "role": item.role,
                    "content": item.message
                }))
            };

            console.log('å¼€å§‹å‘é€æµå¼è¯·æ±‚...');
            const response = await fetch(apiUrl, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(requestData)
            });

            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }

            const reader = response.body.getReader();
            const decoder = new TextDecoder();

            // ç§»é™¤æ‰“å­—åŠ¨ç”»ï¼Œå‡†å¤‡æ˜¾ç¤ºæµå¼å†…å®¹
            removeTypingAnimation();

            // åˆ›å»ºæ–°çš„AIæ¶ˆæ¯å®¹å™¨
            aiMessageElement = createStreamingMessage('ai');
            console.log('å·²åˆ›å»ºæµå¼æ¶ˆæ¯å®¹å™¨');

            let chunkCount = 0;
            while (true) {
                const { done, value } = await reader.read();
                console.log('è¯»å–åˆ°æ•°æ®å—:', { done, valueLength: value?.length });

                if (done) {
                    console.log('æµå¼å“åº”å®Œæˆ');
                    break;
                }

                const chunk = decoder.decode(value);
                console.log('è§£ç åçš„æ•°æ®å—:', chunk);
                const lines = chunk.split('\n');

                for (const line of lines) {
                    if (line.startsWith('data: ')) {
                        try {
                            const data = JSON.parse(line.slice(6));
                            console.log('è§£æåçš„æ•°æ®:', data);

                            if (!data.success) {
                                throw new Error(data.message || 'Stream error');
                            }

                            if (data.content) {
                                chunkCount++;
                                // ç´¯ç§¯æµå¼å†…å®¹
                                accumulatedResponse += data.content;
                                console.log(`æ¥æ”¶åˆ°ç¬¬${chunkCount}ä¸ªæ•°æ®å—ï¼Œå½“å‰æ€»é•¿åº¦:`, accumulatedResponse.length);
                                
                                if (aiMessageElement) {
                                    // æµå¼æ˜¾ç¤ºæ—¶åªè½¬ä¹‰ç‰¹æ®Šå­—ç¬¦ï¼Œä¸è¿›è¡Œmarkdownè½¬æ¢ï¼Œç¡®ä¿æµç•…æ€§
                                    const safeText = accumulatedResponse
                                        .replace(/&/g, '&amp;')
                                        .replace(/</g, '&lt;')
                                        .replace(/>/g, '&gt;')
                                        .replace(/"/g, '&quot;')
                                        .replace(/'/g, '&#39;')
                                        .replace(/\n/g, '<br>');
                                    updateStreamingMessage(aiMessageElement, safeText);
                                    console.log('å·²æ›´æ–°æ¶ˆæ¯æ˜¾ç¤º');
                                }
                            }

                            if (data.finished) {
                                console.log('æµå¼å“åº”ç»“æŸæ ‡å¿—');
                                // ä¿å­˜å®Œæ•´å“åº”åˆ°å¯¹è¯å†å²
                                    content.push({
                                        role: 'assistant',
                                        message: accumulatedResponse,
                                    });
                                    
                                    let timestamp = new Date().toLocaleString();
                                    conversationTranscript.push({
                                        sender: 'bot',
                                        time: timestamp,
                                        message: accumulatedResponse,
                                    });
                                
                                // ä¸ºæœ€ç»ˆæ¶ˆæ¯åº”ç”¨å®Œæ•´markdownæ ¼å¼åŒ–
                                if (aiMessageElement) {
                                    const finalHtml = markdownToHtml(accumulatedResponse);
                                    aiMessageElement.innerHTML = finalHtml;
                                    finalizeStreamingMessage(aiMessageElement, accumulatedResponse);
                                }
                                return;
                            }
                        } catch (parseError) {
                            console.error('è§£ææµå¼æ•°æ®é”™è¯¯:', parseError);
                        }
                    }
                }
            }

        } catch (error) {
            console.error('æµå¼å“åº”é”™è¯¯:', error);
            removeTypingAnimation();
            showPopup('Oops! Something went wrong with streaming!');

            // å¦‚æœæµå¼è¯·æ±‚å¤±è´¥ï¼Œå›é€€åˆ°éæµå¼è¯·æ±‚
            console.log('å›é€€åˆ°éæµå¼è¯·æ±‚...');
            await fetchDataNonStream();
        } finally {
            button.disabled = false;
        }
    }

    function typeWriterEffect(element, text, speed = 20) {
        // æ¸…ç©ºå…ƒç´ å†…å®¹
        element.innerHTML = '';
        
        let i = 0;
        const timer = setInterval(() => {
            if (i < text.length) {
                element.innerHTML = text.substring(0, i + 1);
                i++;
                
                // æ»šåŠ¨åˆ°åº•éƒ¨
                const chatMessagesContainer = document.querySelector('.chat__messages');

                    chatMessagesContainer.scrollTop = chatMessagesContainer.scrollHeight;

            } else {
                clearInterval(timer);
            }
        }, speed);
    }

    function copyText(event) {
        const messageContainer = event.target.closest('.chat__messages__ai, .chat__messages__user');
        const clone = messageContainer.cloneNode(true);
        // ç§»é™¤å¤åˆ¶æŒ‰é’®å’Œæç¤ºæ–‡æœ¬

            elem.parentNode.removeChild(elem);
        });
        clone.querySelectorAll('[style*="display:none"]').forEach(elem => {
            elem.parentNode.removeChild(elem);
        });
        const textToCopy = clone.textContent.trim();
        navigator.clipboard.writeText(textToCopy)
            .then(() => {
                const copiedSpan = event.target.nextElementSibling;
                if (copiedSpan) {
                    copiedSpan.style.display = 'block';
                    setTimeout(() => {
                        copiedSpan.style.display = 'none';
                    }, 2000);
                }
            })
            .catch(error => {
                console.error('Error copying text: ', error);
            });
    }

    async function identifyImage(file) {
        const formData = new FormData();
        formData.append('image', file);
        formData.append('brick_type', 'all'); // é»˜è®¤è¯†åˆ«æ‰€æœ‰ç±»å‹

        try {
            const response = await fetch(imageIdentificationUrl, {
                method: 'POST',
                body: formData
            });

            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }

            const data = await response.json();

            if (data.success) {
                // ç§»é™¤æ‰“å­—åŠ¨ç”»
                removeTypingAnimation();

                // æ£€æŸ¥æ˜¯å¦æ˜¯å›¾ç‰‡è¯†åˆ«ç»“æœ
                let htmlContent;
                if (data.type === 'brick_recognition') {
                    // å›¾ç‰‡è¯†åˆ«ç»“æœï¼Œå…ˆæå–divå†…çš„markdownå†…å®¹ï¼Œè½¬æ¢åå†æ”¾å›divä¸­
                    const divMatch = data.result.match(/<div class="brick-recognition-result">([\s\S]*?)<\/div>/);
                    if (divMatch) {
                        const markdownContent = divMatch[1];
                        const convertedHtml = markdownToHtml(markdownContent);
                        htmlContent = `<div class="brick-recognition-result">${convertedHtml}</div>`;
                    } else {
                        htmlContent = markdownToHtml(data.result);
                    }
                } else {
                    // æ™®é€šèŠå¤©ç»“æœï¼Œä½¿ç”¨markdownè½¬æ¢
                    htmlContent = markdownToHtml(data.result);
                }

                replaceTypingAnimationWithMessage('ai', htmlContent);

                // ä¿å­˜AIå“åº”åˆ°å¯¹è¯å†å²
                content.push({
                    role: 'assistant',
